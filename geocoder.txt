package com.example.mad

import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AppCompatActivity
import android.location.Geocoder
import java.util.*

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val editLocation = findViewById<EditText>(R.id.editLocation)
        val btnGetLatLng = findViewById<Button>(R.id.btnGetLatLng)
        val btnGetAddress = findViewById<Button>(R.id.btnGetAddress)
        val txtResult = findViewById<TextView>(R.id.txtResult)

        val geocoder = Geocoder(this, Locale.getDefault())

        // LOCATION → LAT/LNG
        btnGetLatLng.setOnClickListener {
            val location = editLocation.text.toString()

            try {
                val list = geocoder.getFromLocationName(location, 1)
                if (list != null && list.isNotEmpty()) {
                    val address = list[0]
                    val lat = address.latitude
                    val lng = address.longitude

                    txtResult.text = "Lat: $lat\nLng: $lng"
                } else {
                    txtResult.text = "Location not found"
                }
            } catch (e: Exception) {
                txtResult.text = "Error: ${e.message}"
            }
        }

        // LAT/LNG → ADDRESS (Sample coordinates)
        btnGetAddress.setOnClickListener {
            try {
                val list = geocoder.getFromLocation(9.8808, 78.0700, 1)

                if (list != null && list.isNotEmpty()) {
                    val address = list[0]
                    txtResult.text = address.getAddressLine(0)
                } else {
                    txtResult.text = "Address not found"
                }
            } catch (e: Exception) {
                txtResult.text = "Error: ${e.message}"
            }
        }
    }
}
