package com.example.mad

import android.app.*
import android.content.Context
import android.content.Intent
import android.content.SharedPreferences
import android.net.Uri
import android.os.Build
import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AlertDialog
import androidx.appcompat.app.AppCompatActivity
import androidx.core.app.NotificationCompat
import java.util.*

class MainActivity : AppCompatActivity() {

    private val nameList = ArrayList<String>()
    private lateinit var sharedPref: SharedPreferences

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val editName = findViewById<EditText>(R.id.editName)
        val btnAdd = findViewById<Button>(R.id.btnAdd)
        val btnDelete = findViewById<Button>(R.id.btnDelete)
        val btnPopup = findViewById<Button>(R.id.btnPopup)
        val btnMap = findViewById<Button>(R.id.btnMap)
        val btnNotify = findViewById<Button>(R.id.btnNotify)
        val btnAlert = findViewById<Button>(R.id.btnAlert)
        val btnProgress = findViewById<Button>(R.id.btnProgress)
        val btnDate = findViewById<Button>(R.id.btnDate)
        val btnTime = findViewById<Button>(R.id.btnTime)

        sharedPref = getSharedPreferences("MyApp", MODE_PRIVATE)

        val saved = sharedPref.getString("names", "")
        if (!saved.isNullOrEmpty()) nameList.addAll(saved.split(","))

        // ADD
        btnAdd.setOnClickListener {
            val name = editName.text.toString()
            if (name.isNotEmpty()) {
                nameList.add(name)
                saveData()
                toast("Added: $name")
                editName.text.clear()
            }
        }

        // DELETE
        btnDelete.setOnClickListener {
            val name = editName.text.toString()
            if (nameList.remove(name)) {
                saveData()
                toast("Deleted")
            } else toast("Not found")
        }

        // POPUP
        btnPopup.setOnClickListener {
            val popup = PopupMenu(this, btnPopup)
            popup.menu.add("Show Names")
            popup.setOnMenuItemClickListener {
                toast(nameList.toString())
                true
            }
            popup.show()
        }

        // MAP
        btnMap.setOnClickListener {
            val uri = Uri.parse("geo:0,0?q=Thiruparankundram, Madurai")
            startActivity(Intent(Intent.ACTION_VIEW, uri))
        }

        // NOTIFICATION
        btnNotify.setOnClickListener { showNotification() }

        // ALERT DIALOG
        btnAlert.setOnClickListener {
            AlertDialog.Builder(this)
                .setTitle("Confirm")
                .setMessage("Do you like this app?")
                .setPositiveButton("Yes") { _, _ -> toast("Thank you!") }
                .setNegativeButton("No") { _, _ -> toast("Feedback noted") }
                .show()
        }

        // PROGRESS
        btnProgress.setOnClickListener {
            val progress = ProgressDialog(this)
            progress.setMessage("Loading...")
            progress.setCancelable(false)
            progress.show()

            android.os.Handler().postDelayed({
                progress.dismiss()
                toast("Done")
            }, 3000)
        }

        // DATE PICKER
        btnDate.setOnClickListener {
            val cal = Calendar.getInstance()
            DatePickerDialog(this,
                { _, y, m, d -> toast("$d/${m + 1}/$y") },
                cal.get(Calendar.YEAR),
                cal.get(Calendar.MONTH),
                cal.get(Calendar.DAY_OF_MONTH)
            ).show()
        }

        // TIME PICKER
        btnTime.setOnClickListener {
            val cal = Calendar.getInstance()
            TimePickerDialog(this,
                { _, h, m -> toast("$h:$m") },
                cal.get(Calendar.HOUR_OF_DAY),
                cal.get(Calendar.MINUTE),
                true
            ).show()
        }
    }

    private fun saveData() {
        sharedPref.edit().putString("names", nameList.joinToString(",")).apply()
    }

    private fun toast(msg: String) {
        Toast.makeText(this, msg, Toast.LENGTH_SHORT).show()
    }

    private fun showNotification() {
        val channelId = "demo_channel"
        val manager = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val channel = NotificationChannel(channelId, "Demo", NotificationManager.IMPORTANCE_DEFAULT)
            manager.createNotificationChannel(channel)
        }

        val notification = NotificationCompat.Builder(this, channelId)
            .setContentTitle("Hello")
            .setContentText("Names saved successfully")
            .setSmallIcon(android.R.drawable.ic_dialog_info)
            .build()

        manager.notify(1, notification)
    }
}
