package com.example.mad

import android.app.NotificationChannel
import android.app.NotificationManager
import android.content.Context
import android.content.Intent
import android.content.SharedPreferences
import android.net.Uri
import android.os.Build
import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AppCompatActivity
import androidx.core.app.NotificationCompat

class MainActivity : AppCompatActivity() {

    private val nameList = ArrayList<String>()
    private lateinit var sharedPref: SharedPreferences

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val editName = findViewById<EditText>(R.id.editName)
        val btnAdd = findViewById<Button>(R.id.btnAdd)
        val btnDelete = findViewById<Button>(R.id.btnDelete)
        val btnPopup = findViewById<Button>(R.id.btnPopup)
        val btnMap = findViewById<Button>(R.id.btnMap)
        val btnNotify = findViewById<Button>(R.id.btnNotify)

        // SharedPreferences init
        sharedPref = getSharedPreferences("MyApp", MODE_PRIVATE)

        // LOAD saved data
        val savedData = sharedPref.getString("names", "")
        if (!savedData.isNullOrEmpty()) {
            nameList.addAll(savedData.split(","))
        }

        // ADD NAME
        btnAdd.setOnClickListener {
            val name = editName.text.toString()
            if (name.isNotEmpty()) {
                nameList.add(name)
                saveData()
                toast("Added: $name")
                editName.text.clear()
            } else {
                toast("Enter name")
            }
        }

        // DELETE NAME
        btnDelete.setOnClickListener {
            val name = editName.text.toString()
            if (nameList.remove(name)) {
                saveData()
                toast("Deleted: $name")
            } else {
                toast("Name not found")
            }
        }

        // POPUP MENU
        btnPopup.setOnClickListener {
            val popup = PopupMenu(this, btnPopup)
            popup.menu.add("Show All Names")

            popup.setOnMenuItemClickListener {
                toast(nameList.toString())
                true
            }

            popup.show()
        }

        // MAP → Thiruparankundram
        btnMap.setOnClickListener {
            val uri = Uri.parse("geo:0,0?q=Thiruparankundram, Madurai")
            val intent = Intent(Intent.ACTION_VIEW, uri)
            intent.setPackage("com.google.android.apps.maps")
            startActivity(intent)
        }

        // NOTIFICATION
        btnNotify.setOnClickListener {
            showNotification()
        }
    }

    private fun saveData() {
        val editor = sharedPref.edit()
        val data = nameList.joinToString(",")
        editor.putString("names", data)
        editor.apply()
    }

    private fun toast(msg: String) {
        Toast.makeText(this, msg, Toast.LENGTH_SHORT).show()
    }

    private fun showNotification() {
        val channelId = "demo_channel"
        val manager = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val channel = NotificationChannel(
                channelId,
                "Demo Notifications",
                NotificationManager.IMPORTANCE_DEFAULT
            )
            manager.createNotificationChannel(channel)
        }

        val notification = NotificationCompat.Builder(this, channelId)
            .setContentTitle("Hello 👋")
            .setContentText("Your saved names are محفوظ (stored)")
            .setSmallIcon(android.R.drawable.ic_dialog_info)
            .build()

        manager.notify(1, notification)
    }
}
