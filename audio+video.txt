package com.example.mad

import android.media.MediaPlayer
import android.net.Uri
import android.os.Bundle
import android.view.View
import android.view.animation.*
import android.widget.Button
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import android.content.Intent
import android.widget.VideoView

class MainActivity : AppCompatActivity() {

    private var mediaPlayer: MediaPlayer? = null

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val btnAudio = findViewById<Button>(R.id.btnAudio)
        val btnVideo = findViewById<Button>(R.id.btnVideo)

        val box = findViewById<View>(R.id.box)
        val btnMove = findViewById<Button>(R.id.btnMove)
        val btnRotate = findViewById<Button>(R.id.btnRotate)
        val btnScale = findViewById<Button>(R.id.btnScale)
        val btnFade = findViewById<Button>(R.id.btnFade)

        // AUDIO
        btnAudio.setOnClickListener {
            mediaPlayer = MediaPlayer.create(this, R.raw.sample_audio)
            mediaPlayer?.start()
            toast("Playing Audio")
        }

        // VIDEO
        val videoView = findViewById<VideoView>(R.id.videoView)

        btnVideo.setOnClickListener {
            val uri = Uri.parse("android.resource://" + packageName + "/" + R.raw.sample_video)
            videoView.setVideoURI(uri)
            videoView.start()
            toast("Playing Video")
        }
        // MOVE
        btnMove.setOnClickListener {
            val anim = TranslateAnimation(0f, 300f, 0f, 0f)
            anim.duration = 1000
            box.startAnimation(anim)
        }

        // ROTATE
        btnRotate.setOnClickListener {
            val anim = RotateAnimation(
                0f, 360f,
                Animation.RELATIVE_TO_SELF, 0.5f,
                Animation.RELATIVE_TO_SELF, 0.5f
            )
            anim.duration = 1000
            box.startAnimation(anim)
        }

        // SCALE
        btnScale.setOnClickListener {
            val anim = ScaleAnimation(
                1f, 2f, 1f, 2f,
                Animation.RELATIVE_TO_SELF, 0.5f,
                Animation.RELATIVE_TO_SELF, 0.5f
            )
            anim.duration = 1000
            box.startAnimation(anim)
        }

        // FADE
        btnFade.setOnClickListener {
            val anim = AlphaAnimation(1f, 0f)
            anim.duration = 1000
            box.startAnimation(anim)
        }
    }

    private fun toast(msg: String) {
        Toast.makeText(this, msg, Toast.LENGTH_SHORT).show()
    }
}
