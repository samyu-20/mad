package com.example.mad

import android.content.ContentValues
import android.content.Context
import android.database.sqlite.SQLiteDatabase
import android.database.sqlite.SQLiteOpenHelper

class DBHelper(context: Context) :
    SQLiteOpenHelper(context, "NameDB", null, 1) {

    override fun onCreate(db: SQLiteDatabase) {
        db.execSQL("CREATE TABLE names(id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT)")
    }

    override fun onUpgrade(db: SQLiteDatabase, oldVersion: Int, newVersion: Int) {
        db.execSQL("DROP TABLE IF EXISTS names")
        onCreate(db)
    }

    // INSERT
    fun insertName(name: String): Boolean {
        val db = this.writableDatabase
        val cv = ContentValues()
        cv.put("name", name)
        val result = db.insert("names", null, cv)
        return result != -1L
    }

    // DELETE
    fun deleteName(name: String): Boolean {
        val db = this.writableDatabase
        val result = db.delete("names", "name=?", arrayOf(name))
        return result > 0
    }

    // FETCH ALL
    fun getAllNames(): ArrayList<String> {
        val list = ArrayList<String>()
        val db = this.readableDatabase
        val cursor = db.rawQuery("SELECT * FROM names", null)

        if (cursor.moveToFirst()) {
            do {
                list.add(cursor.getString(1))
            } while (cursor.moveToNext())
        }
        cursor.close()
        return list
    }
}

package com.example.mad

import android.app.NotificationChannel
import android.app.NotificationManager
import android.content.Context
import android.content.Intent
import android.net.Uri
import android.os.Build
import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AppCompatActivity
import androidx.core.app.NotificationCompat

class MainActivity : AppCompatActivity() {

    lateinit var db: DBHelper

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val editName = findViewById<EditText>(R.id.editName)
        val btnAdd = findViewById<Button>(R.id.btnAdd)
        val btnDelete = findViewById<Button>(R.id.btnDelete)
        val btnPopup = findViewById<Button>(R.id.btnPopup)
        val btnMap = findViewById<Button>(R.id.btnMap)
        val btnNotify = findViewById<Button>(R.id.btnNotify)

        db = DBHelper(this)

        // ADD
        btnAdd.setOnClickListener {
            val name = editName.text.toString()
            if (name.isNotEmpty()) {
                if (db.insertName(name)) {
                    toast("Added: $name")
                    editName.text.clear()
                } else {
                    toast("Error adding name")
                }
            } else {
                toast("Enter name")
            }
        }

        // DELETE
        btnDelete.setOnClickListener {
            val name = editName.text.toString()
            if (db.deleteName(name)) {
                toast("Deleted: $name")
            } else {
                toast("Name not found")
            }
        }

        // POPUP → SHOW ALL
        btnPopup.setOnClickListener {
            val popup = PopupMenu(this, btnPopup)
            popup.menu.add("Show All Names")

            popup.setOnMenuItemClickListener {
                val list = db.getAllNames()
                toast(list.toString())
                true
            }

            popup.show()
        }

        // MAP
        btnMap.setOnClickListener {
            val uri = Uri.parse("geo:0,0?q=Thiruparankundram, Madurai")
            val intent = Intent(Intent.ACTION_VIEW, uri)
            intent.setPackage("com.google.android.apps.maps")
            startActivity(intent)
        }

        // NOTIFICATION
        btnNotify.setOnClickListener {
            showNotification()
        }
    }

    private fun toast(msg: String) {
        Toast.makeText(this, msg, Toast.LENGTH_SHORT).show()
    }

    private fun showNotification() {
        val channelId = "demo_channel"
        val manager = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val channel = NotificationChannel(channelId, "Demo", NotificationManager.IMPORTANCE_DEFAULT)
            manager.createNotificationChannel(channel)
        }

        val notification = NotificationCompat.Builder(this, channelId)
            .setContentTitle("SQLite Data")
            .setContentText("Names stored successfully")
            .setSmallIcon(android.R.drawable.ic_dialog_info)
            .build()

        manager.notify(1, notification)
    }
}
